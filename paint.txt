#include <vector> // [NOWE] Do przechowywania listy elips

// [NOWE] Struktura opisująca elipsę
struct ellipse_data {
    RECT rect;
    COLORREF color;
};

class app_2048 {
private:
    std::vector<ellipse_data> m_ellipses; // [NOWE] Lista wszystkich elips
    POINT m_start_drag;                   // [NOWE] Punkt rozpoczęcia rysowania
    bool m_is_dragging = false;           // [NOWE] Czy użytkownik trzyma lewy przycisk

    // ... reszta pól ...
    void draw_all_ellipses(HDC hdc);      // [NOWE] Metoda pomocnicza
};

void app_2048::draw_all_ellipses(HDC hdc) {
    for (const auto& ellipse : m_ellipses) {
        // Tworzymy pędzel w kolorze elipsy
        HBRUSH brush = CreateSolidBrush(ellipse.color);
        HBRUSH old_brush = (HBRUSH)SelectObject(hdc, brush);
        
        // Rysujemy elipsę w zapisanym prostokącie
        Ellipse(hdc, ellipse.rect.left, ellipse.rect.top, ellipse.rect.right, ellipse.rect.bottom);
        
        // Sprzątamy zasoby GDI
        SelectObject(hdc, old_brush);
        DeleteObject(brush);
    }
}

void app_2048::draw_gradient(HWND window) {
    // ... (kod rysowania gradientu bez zmian) ...
    GradientFill(hdc, vertex, 2, &g_rect, 1, GRADIENT_FILL_RECT_H);

    // [NOWE] Po narysowaniu tła, rysujemy na nim elipsy
    draw_all_ellipses(hdc);

    EndPaint(window, &ps);
}

LRESULT app_2048::window_proc(HWND window, UINT message, WPARAM wparam, LPARAM lparam) {
    switch (message) {
    case WM_LBUTTONDOWN: // Kliknięcie - zaczynamy rysować
        m_start_drag.x = LOWORD(lparam);
        m_start_drag.y = HIWORD(lparam);
        m_is_dragging = true;
        SetCapture(window); // Przechwyć mysz poza oknem
        return 0;

    case WM_LBUTTONUP: // Puszczenie - zapisujemy elipsę na stałe
        if (m_is_dragging) {
            ellipse_data new_ellipse;
            new_ellipse.rect = { m_start_drag.x, m_start_drag.y, LOWORD(lparam), HIWORD(lparam) };
            new_ellipse.color = m_selected_color; // Używamy aktualnie wybranego koloru z menu!
            
            m_ellipses.push_back(new_ellipse);
            m_is_dragging = false;
            ReleaseCapture();
            InvalidateRect(window, nullptr, TRUE); // Odśwież, by zobaczyć nową elipsę
        }
        return 0;

    case WM_MOUSEMOVE: // Przesuwanie - opcjonalnie: tutaj można dodać podgląd "na żywo"
        if (m_is_dragging) {
            // Dla prostoty odświeżamy okno tylko przy puszczeniu, 
            // ale InvalidateRect tutaj dałoby podgląd "rozciągania" elipsy.
        }
        return 0;
    
    // ... reszta case-ów (WM_COMMAND, WM_PAINT, itd.) ...
    }
}
