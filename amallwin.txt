LRESULT CALLBACK app_2048::sub_window_proc(HWND window, UINT message, WPARAM wparam, LPARAM lparam) {
    if (message == WM_PAINT) {
        PAINTSTRUCT ps;
        HDC hdc = BeginPaint(window, &ps);
        
        RECT rect;
        GetClientRect(window, &rect);

        // Pobieramy wskaŸnik do g³ównej aplikacji, aby znaæ wybrany kolor
        // (Wymaga to drobnej poprawki w CreateWindowEx powy¿ej, by przekazaæ 'this', 
        // ale dla uproszczenia mo¿emy u¿yæ GetParent i wyci¹gn¹æ kolor lub u¿yæ zmiennej globalnej/statycznej)
        
        TRIVERTEX vertex[2];
        vertex[0].x = 0;
        vertex[0].y = 0;
        vertex[0].Red = 0xFFFF;   // Bia³y start
        vertex[0].Green = 0xFFFF;
        vertex[0].Blue = 0xFFFF;
        
        vertex[1].x = rect.right;
        vertex[1].y = rect.bottom;
        vertex[1].Red = 0x0000;   // Czarny koniec
        vertex[1].Green = 0x0000;
        vertex[1].Blue = 0x0000;

        GRADIENT_RECT g_rect{ 0, 1 };
        GradientFill(hdc, vertex, 2, &g_rect, 1, GRADIENT_FILL_RECT_H);

        EndPaint(window, &ps);
        return 0;
    }
    return DefWindowProcW(window, message, wparam, lparam);
}
